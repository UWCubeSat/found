65inline EulerAngles strtoea(const std::string &str) {
66 char delimiter = str.find(" ") != std::string::npos ? ' ' : ',';
67 decimal result[3];
68
69 size_t start = 0;
70 size_t end = str.find(delimiter);
71 size_t index = 0;
72
73 while (index != 3 && end != std::string::npos) {
74 result[index++] = strtodecimal(str.substr(start, end - start));
75 start = end + 1;
76 end = str.find(delimiter, start);

127inline LocationRecords strtolr(const std::string &str) {
128 LocationRecords records;
129 // TODO: If the file is a Data File, then parse it differently
130 std::ifstream file(str);
131 if (!file.is_open()) {
132 throw std::runtime_error("Could not open file " + str);
133 }
134
135 std::string line;
136 while (std::getline(file, line)) {
137 std::istringstream iss(line);
138 LocationRecord record;
139 if (!(iss >> record.timestamp >> record.position.x >> record.position.y >> record.position.z)) {
140 file.close();
141 throw std::runtime_error("Invalid format in file " + str + ": " + line);
142 }
143 records.push_back(record);
144 }
145
146 file.close();
147 return records;
148}
