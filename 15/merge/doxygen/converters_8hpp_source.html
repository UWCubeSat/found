67inline EulerAngles strtoea(const std::string &str) {
68 char delimiter = str.find(" ") != std::string::npos ? ' ' : ',';
69 decimal result[3];
70
71 size_t start = 0;
72 size_t end = str.find(delimiter);
73 size_t index = 0;
74
75 while (index != 2 && end != std::string::npos) {
76 result[index++] = strtodecimal(str.substr(start, end - start));

78 end = str.find(delimiter, start);

139inline LocationRecords strtolr(const std::string &str) {
140 if (str.size() >= 6) {
141 if (str.substr(str.size() - 6) == ".found") {
142 LOG_INFO("Getting Position Data from Data File (*.found)");
143 DataFile data = strtodf(str);
144 return LocationRecords(data.positions.get(), data.positions.get() + data.header.num_positions);
145 }
146 }
147
148 LOG_INFO("Getting Position Data from non-Data File (not *.found)");
149 LocationRecords records;
150 std::ifstream file(str);
151 if (!file.is_open()) {
152 throw std::runtime_error("Could not open file " + str);
153 }
154
155 std::string line;
156 while (std::getline(file, line)) {
157 std::istringstream iss(line);
158 LocationRecord record;
159 if (!(iss >> record.timestamp >> record.position.x >> record.position.y >> record.position.z)) {
160 file.close();
161 throw std::runtime_error("Invalid format for file " + str + ": " + line);
162 }
163 records.push_back(record);
164 }
165
166 file.close();
167 return records;
168}
