75inline EulerAngles strtoea(const std::string &str) {
76 char delimiter = str.find(" ") != std::string::npos ? ' ' : ',';
77 decimal result[3];
78
79 size_t start = 0;
80 size_t end = str.find(delimiter);
81 size_t index = 0;
82
83 while (index != 2 && end != std::string::npos) {
84 result[index++] = strtodecimal(str.substr(start, end - start));
85 start = end + 1;
86 end = str.find(delimiter, start);
87 }
88
89 result[index++] = strtodecimal(str.substr(start));
90
91 while (index != 3) result[index++] = 0;
92
93 return EulerAngles(DegToRad(result[0]), DegToRad(result[1]), DegToRad(result[2]));
94}

135inline DateTime strtodatetime(const std::string &str) {
136 std::tm tm = {};
137 std::istringstream ss(str);
138
139 ss >> std::get_time(&tm, "%Y-%m-%d %H:%M:%S");
140 if (ss.fail()) {
141 throw std::invalid_argument("Invalid datetime format: " + str);
142 }
143
144 // Store original values for validation
145 uint64_t year = static_cast<uint64_t>(tm.tm_year + 1900);
146 uint64_t month = static_cast<uint64_t>(tm.tm_mon + 1);
147 uint64_t day = static_cast<uint64_t>(tm.tm_mday);
148 uint64_t hour = static_cast<uint64_t>(tm.tm_hour);
149 uint64_t minute = static_cast<uint64_t>(tm.tm_min);
150 uint64_t second = static_cast<uint64_t>(tm.tm_sec);
151
152 if (second > 59) {
153 throw std::invalid_argument("Invalid second in datetime: " + str);
154 }
155
156 // Validate day of month (considering leap years)
157 bool is_leap_year = (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0);
158 uint64_t max_days = static_cast<uint64_t>(days_in_month[month - 1]);
159 if (is_leap_year && month == 2) {
160 max_days = 29;
161 }
162 if (day > max_days) {
163 throw std::invalid_argument("Invalid day in datetime: " + str);
164 }
165
166 std::time_t t = timegm(&tm);
167
168 uint64_t nanosecond = 0;
169 std::string nanos_str;
170 if (std::getline(ss, nanos_str) && nanos_str.size() > 1 && nanos_str[0] == '.') {
171 nanos_str = nanos_str.substr(1);
172 nanos_str.resize(9, '0'); // pad or truncate to 9 digits
173 nanosecond = static_cast<uint64_t>(std::stoul(nanos_str));
174 }
175
176 // Convert to nanoseconds: seconds * NS_PER_SEC + nanoseconds
177 uint64_t epochs_ns = static_cast<uint64_t>(t) * NS_PER_SEC + nanosecond;
178
179 return {
180 epochs_ns,
181 year,
182 month,
183 day,
184 hour,
185 minute,
186 second,
187 nanosecond
188 };
189}

210inline LocationRecords strtolr(const std::string &str) {
211 if (str.size() >= 6) {
212 if (str.substr(str.size() - 6) == ".found") {
213 LOG_INFO("Getting Position Data from Data File (*.found)");
214 DataFile data = strtodf(str);
215 return LocationRecords(data.positions.get(), data.positions.get() + data.header.num_positions);
216 }
217 }
218
219 LOG_INFO("Getting Position Data from non-Data File (not *.found)");
220 LocationRecords records;
221 std::ifstream file(str);
222 if (!file.is_open()) {
223 throw std::runtime_error("Could not open file " + str);
224 }
225
226 std::string line;
227 while (std::getline(file, line)) {
228 std::istringstream iss(line);
229 LocationRecord record;
230 if (!(iss >> record.timestamp >> record.position.x >> record.position.y >> record.position.z)) {
231 file.close();
232 throw std::runtime_error("Invalid format for file " + str + ": " + line);
233 }
234 records.push_back(record);
235 }
236
237 file.close();
238 return records;
239}
