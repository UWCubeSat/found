63inline EulerAngles strtoea(const std::string &str) {
64 char delimiter = str.find(" ") != std::string::npos ? ' ' : ',';
65 decimal result[3];
66
67 size_t start = 0;
68 size_t end = str.find(delimiter);
69 size_t index = 0;
70
71 while (index != 3 && end != std::string::npos) {
72 result[index++] = strtodecimal(str.substr(start, end - start));
73 start = end + 1;
74 end = str.find(delimiter, start);
75 }
76
77 result[index] = strtodecimal(str.substr(start));
78
79 return EulerAngles(result[0], result[1], result[2]);
80}

125inline LocationRecords strtolr(const std::string &str) {
126 LocationRecords records;
127 // TODO: If the file is a Data File, then parse it differently
128 std::ifstream file(str);
129 if (!file.is_open()) {
130 throw std::runtime_error("Could not open file " + str);
131 }
132
133 std::string line;
134 while (std::getline(file, line)) {
135 std::istringstream iss(line);
136 LocationRecord record;
137 if (!(iss >> record.timestamp >> record.position.x >> record.position.y >> record.position.z)) {
138 file.close();
139 throw std::runtime_error("Invalid format in file " + str + ": " + line);
140 }
141 records.push_back(record);
142 }
143
144 file.close();
145 return records;
146}
