176 SequentialPipeline &AddStage(std::unique_ptr<StageType> stage) {
177 static_assert(std::is_base_of<FunctionStage<typename StageType::InputType,
178 typename StageType::OutputType>, StageType>::value,
179 "Stage must derive from FunctionStage");
180 StageType *stagePtr = stage.get();
181 if (this->size == 0) {
182 if (!std::is_same<Input, typename StageType::InputType>::value) {
183 throw std::invalid_argument("The initial input type is not correct");
184 }
185 this->firstResource = reinterpret_cast<Input *>(&stagePtr->GetResource());

190 // Add to our list

192 // Now, reset the lastProduct to be of this stage

194 // Return the pipeline for chaining

215 static_assert(std::is_same<Output, typename StageType::OutputType>::value,
216 "Final stage output must match pipeline output");
217 assert(this->size < N);
218 if (this->ready) throw std::invalid_argument("Pipeline is already ready");
219 this->AddStage(std::move(stage));
220 this->ready = true;

240 Output Run(const Input &input) override {
241 assert(!this->finalProduct);
242 // MANUAL VERIFICATION: The below branch is fully tested via pipeline-test.cpp
243 if (!this->ready)
244 throw std::runtime_error("This is an illegal action: the pipeline is not ready yet");
245 // Don't use "Input resource" unless necessary
246 *this->firstResource = input;
247 // If this pipeline is not composed, construct output
248 // into ourself, otherwise construct into the outer pipeline
249 if (this->product == nullptr) {
250 // engage our finalProduct
251 this->finalProduct.emplace();
252 *this->lastProduct = &this->finalProduct.value();
253 this->product = &this->finalProduct.value();
254 } else {
255 *this->lastProduct = this->product;
256 }
257 Pipeline<Input, Output, N>::DoActionHelper();

259 }
